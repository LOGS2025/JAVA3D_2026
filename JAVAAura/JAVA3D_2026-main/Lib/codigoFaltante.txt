    @Override

    }

      this.addKeyListener(new KeyAdapter() {
          @Override
          public void keyPressed(KeyEvent e) {
              if(e.getKeyChar() == 'w'){
                  lienzo.mueveAdelante();
              }
              if(e.getKeyChar() == 's'){
                  lienzo.mueveAtras();
              }

              lienzo.paintComponent(getGraphics());
          }
      });
